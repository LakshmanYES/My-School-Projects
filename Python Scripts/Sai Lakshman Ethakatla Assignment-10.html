def get_statistics_report(df_score):
    "To prompt the user to give input"
    dept_chair_input = input(f'Enter the grade to display data: ')
    
    df_dept_input_grade = df_score[dept_chair_input].count()
    
    
    total_count = df_score[dept_chair_input].count()
    

    mean_average = df_score[dept_chair_input].mean()
    to_round_avg = round(mean_average,1)
    
    score_median = df_score[dept_chair_input].median()
    to_round_median = round(score_median,1)
    
    
    score_max = df_score[dept_chair_input].max()
    
    score_min = df_score[dept_chair_input].min()
    
    return total_count, mean_average, score_median, score_max, score_min

def get_stand_alone_graph(df_primary_data):

    import matplotlib
    import matplotlib.pyplot as plt
    %matplotlib inline
    from matplotlib.ticker import (MultipleLocator, AutoMinorLocator)


    fig3 = plt.figure(dpi=120)

    fig3, (ax1) = plt.subplots(1,1, figsize=(30,8), sharey=True)


   

    ax1.set_title('Stand_alone_graph', fontsize =16)
    ax1.set_xlabel('Semester', fontsize=18)
    ax1.set_ylabel('Three Letter Grades', fontsize=18)

    ax1.tick_params(axis='y', labelsize=15, direction ='out', length=4)
    ax1.tick_params(axis='x', direction='out', length=4, which= 'both', labelsize=15)

    ax1.xaxis.set_minor_locator(AutoMinorLocator(4))
    ax1.minorticks_on



    max_yvalue = max(df_primary_data['A'].max(), df_primary_data['B'].max(), df_primary_data['C'].max())
    ax1.set_ylim(0,max_yvalue+5)


 
    xaxis = []       

    for val in range(1, len(df_primary_data)+1,1):      
        xaxis.append(val)

    ax1.set_xlim(0, max(xaxis)+1) 
    ax1.grid(True, which='both', linewidth=2, color='white')
    ax1.plot(xaxis, df_primary_data['A'], label= 'A', color='purple', marker='D', markersize=10) 
    ax1.plot(xaxis, df_primary_data['B'], label= 'B', color='blue', marker='d', markersize=10, linestyle = '-.')
    ax1.plot(xaxis, df_primary_data['C'], label= 'C', color='black', marker='o', markersize=10, linestyle='dotted')
    ax1.legend(loc='upper left')
